DEVICE:
SWITCH SW1 = 1;
SWITCH SW2 = 1;
CLOCK CL1 = 3;
CLOCK CL2 = 4;
XOR G1;
NAND G2 = 2;
DTYPE G3;
AND G4 = 2;
NOR G5 = 2;

CONNECTION:
CON SW1 -> G1.I1;
CON SW2 -> G2.I2;
CON CL1 -> G1.I2;
CON CL1 -> G2.I1;
CON CL2 -> G3.CLK;
CON G1 -> G3.DATA;
CON SW1 -> G3.SET;
CON SW2 -> G3.CLEAR;
CON G3.Q -> G4.I2;
CON G3.QBAR -> G5.I1;
CON SW1 -> G4.I1;
CON G2 -> G5.I2;

MONITOR:
MON G4;
MON G5;
